 65:     public static function formatTrace($traceArr) {
 66:         $str = '';
 67:         foreach ($traceArr as $key => $value) {
 68:             $str .= "<li>#{$key} ";
 69:             $str .= isset($value['file']) ? $value['file'] : '';
 70:             $str .= isset($value['line']) ? "({$value['line']}): " : '';
 71:             $str .= isset($value['class']) ? $value['class'] : '';
 72:             $str .= isset($value['type']) ? $value['type'] : '';
 73:             if ($value['function'] == 'unknown') {
 74:                 $value['function'] = 'main';
 75:             }
 76:             $str .= isset($value['function']) ? "{$value['function']}(" : '';
 77:             if (isset($value['args'])) {
 78:                 foreach ($value['args'] as $arg) {
 79:                     if (is_string($arg)) {
 80:                         $str .= "<span class='ToKnotDebugArgs' title='String(" . strlen($arg) . ") " . $arg . "'><b>String(</b>" . substr($arg, 0, 100) . "<b>)</b></span>";
 81:                     } elseif (is_int($arg)) {
 82:                         $str .= "<span class='ToKnotDebugArgs'><b>Integer(</b>" . substr($arg, 0, 100) . "<b>)</b></span>";
 83:                     } elseif (is_float($arg)) {
 84:                         $str .= "<span class='ToKnotDebugArgs'><b>Float(</b>" . substr($arg, 0, 100) . "<b>)</b></span>";
 85:                     } elseif (is_array($arg)) {
 86:                         $str .= "<span class='ToKnotDebugArgs' title='" . print_r($arg, true) . "'><b>Array()</b></span>";
 87:                     } elseif (is_object($arg)) {
 88:                         $str .= "<span class='ToKnotDebugArgs' title='" . print_r($arg, true) . "'><b>Object(</b> " . get_class($arg) . " <b>)</b></span>";
 89:                     } elseif (is_resource($arg)) {
 90:                         $str .= "<span calss='ToKnotDebugArgs'" > print_r($arg, true) . '</span>';
 91:                     }
 92:                 }
 93:             }
 94:             $str .= isset($value['function']) ? ")" : '';
 95:             $str .= "</li>\n";
 96:         }
 97:         return $str;
