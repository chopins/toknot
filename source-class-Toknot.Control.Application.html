202:      * @param string $appNameSpace  Application of Namespace with top without full namespace
203:      * @param string $appPath   Application of directory with full path, and not is Controller layer full path
204:      * @param string $defaultInvoke  The parameter of default invoke class for router when no request uri,
205:      *                                if it is not set,will throw BadClassCallException when user request site root 
206:      *                                and no query,The class name of default with not full namespace
207:      *                                class name can not contain application top namespace and
208:      *                                Controller layer namespace
209:      * @param mixed $_   Variable list of router need of paramers on runtime, the toknot default router support
210:      *                    2 parameters:
211:      *                    $routerMode  use set {@see Router::ROUTER_PATH} or {@see Router::ROUTER_GET_QUERY}
212:      *                    $routerDepth it is controller max namespace level
213:      * @throws BadNamespaceException
214:      * @throws BadClassCallException
215:      * @throws StandardException
216:      */
217:     public function run($appNameSpace, $appPath, $defaultInvoke = '\Index') {
218:         $root = substr($appNameSpace, 0, 1);
219:         try {
220:             if ($root != '\\') {
221:                 throw new BadNamespaceException($appNameSpace);
222:             }
223:             if (!class_exists($this->routerName, true)) {
224:                 throw new BadClassCallException($this->routerName);
225:             }
226:             $routerReflection = new ReflectionClass($this->routerName);
227:             if (!$routerReflection->implementsInterface('Toknot\Control\RouterInterface')) {
228:                 throw new StandardException('Router not support');
229:             }
230:             if ($routerReflection->hasMethod('singleton')) {
231:                 $routerName = $this->routerName;
232:                 $router = $routerName :: singleton();
233:             } else {
234:                 $router = new $this->routerName;
235:             }
236: 
237:             $args = func_get_args();
238:             $this->addAppPath($appPath);
239:             $context = FMAI::singleton($appPath);
240:             call_user_func_array(array($router,'runtimeArgs'), array_slice($args,2));
241:             $router->routerSpace($appNameSpace);
242:             $router->routerPath($appPath);
243:             $router->routerRule();
244:             if(is_null($defaultInvoke)) {
245:                 $root = substr($defaultInvoke, 0, 1);
246:                 if ($root != '\\') {
247:                     throw new BadNamespaceException($defaultInvoke);
248:                 }
249:                 $router->defaultInvoke($defaultInvoke);
250:             }
251:             $router->invoke($context);
252:         } catch (StandardException $e) {
253:             echo $e;
254:         }
255:     }
