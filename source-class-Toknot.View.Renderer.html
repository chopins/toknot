113:             if (file_exists($this->htmlCacheFile)) {

139:     private function transfromToPHP() {
140:         $content = file_get_contents($this->tplName);
141: 
142:         //transfrom variable
143:         $content = preg_replace_callback('/\{\$(\.a-zA-Z0-9_\x7f-\xff]+)\}/i', function($matches) {
144:                     $str = '<?php echo $this->varList->';
145:                     $str .= str_replace('.', '->', $matches[1]);
146:                     $str .= ';?>';
147:                     return $str;
148:                 }, $content);
149: 
150:         //transfrom foreach statement
151:         $content = preg_replace_callback('/\{foreach\s+\$(\S+)\s+as(.*)\}/i', function($matches) {
152:                     $matches[1] = str_replace('.', '->', $matches[1]);
153:                     $str = "<?php if(is_array(\$this->varList->{$matches[1]}) && !empty(\$this->varList->{$matches[1]})){";
154:                     $str .= "foreach(\$this->varList->$matches[1] as $matches[2]) { ?>";
155:                     return $str;
156:                 }, $content);

159:         //transfrom define variable which is not controller set

162:         //transfrom if statement
163:         $content = preg_replace_callback('/\{if\s+([^\{^\}]+)\}/i', function($matches) {
