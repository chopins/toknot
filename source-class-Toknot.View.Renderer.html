122:     public function display($tplName) {
123:         $tplFile = self::$scanPath . '/' . $tplName . '.' . self::$fileExtension;
124:         if (!file_exists($tplFile)) {
125:             throw new StandardException("{$tplFile} not exists");
126:         }
127:         $useCache = false;
128:         //HTML cache control
129:         if (self::$enableCache && self::$htmlCachePath != null) {
130:             $key = md5($_SERVER['QUERY_STRING']);
131:             if (self::$cacheFlag == self::CACHE_FLAG_HTML) {
132:                 $htmlCacheFile = self::$htmlCachePath . '/' . $tplName . '.' . $key . '.html';
133:                 if (file_exists($htmlCacheFile)) {
134:                     $mtime = filemtime($htmlCacheFile);
135:                     if ($mtime + (string) self::$outCacheThreshold <= time()) {
136:                         include_once $htmlCacheFile;
137:                         return ViewCache::CACHE_USE_SUCC;
138:                     }
139:                     ob_start();
140:                 }
141:             } elseif (self::$cacheFlag == self::CACHE_FLAG_DATA) {
142:                 $dataCacheFile = self::$dataCachePath . '/' . $tplName . '.' . $key;
143:                 $cache = new DataCacheControl($dataCacheFile);
144:                 $cache->useExpire(self::$outCacheThreshold);
145:                 $varList = $cache->get();
146:                 if ($varList === false) {
147:                     $cacheData = $this->varList->transformToArray();
148:                     $cache->save($cacheData);
149:                 } else {
150:                     $cacheVar = new ArrayObject($varList);
151:                     $cacheVar->importPropertie($this->varList);
152:                     $this->varList = $cacheVar;
153:                     uset($cacheVar, $varList);
154:                     $useCache = true;
155:                 }
156:             }
157:         }
158: 
159:         $transfromFile = self::$cachePath . '/' . $tplName . '.php';
160:         if (DEVELOPMENT || !file_exists($transfromFile) ||
161:                 filemtime($transfromFile) < filemtime($tplFile)) {
162:             $this->transfromToPHP($tplFile, $transfromFile);
163:         }
164: 
165:         include_once $transfromFile;
166: 
167:         //HTML Cache write
168:         if (self::$enableCache && self::$htmlCachePath != null && self::$cacheFlag == self::CACHE_FLAG_HTML) {
169:             $html = ob_get_contents();
170:             ob_flush();
171:             ob_end_clean();
172:             FileObject::saveContent($htmlCacheFile, $html);
173:         }
174:         if ($useCache) {
175:             return ViewCache::CACHE_USE_SUCC;
176:         }
177:     }

179:     private function transfromToPHP($tplFile, $transfromFile) {
180:         $content = file_get_contents($tplFile);
181: 
182:         //transfrom foreach statement
183:         $content = preg_replace_callback('/\{foreach\040+\$(\S+)\040+as\040+([\$a-zA-Z0-9_=>\040]+)}(.*)\{\/foreach\}/ims', function($matches) {
184:                     $matches[1] = str_replace('.', '->', $matches[1]);
185:                     $str = "<?php foreach(\$this->varList->$matches[1] as $matches[2]) { ?>";
186:                     if (preg_match('/\$([a-zA-Z0-9_]+)\040*=>\040*\$([a-zA-Z0-9_]+)/i', $matches[2], $setValue)) {
187:                         $varName = $setValue[2];
188:                         $keyName = $setValue[1];
189:                     } elseif (preg_match('/\$([a-zA-Z0-9_]+)/i', $matches[2], $setValue)) {
190:                         $varName = $setValue[1];
191:                         $keyName = null;
192:                     }
193:                     $str .= preg_replace_callback('/\{\$([a-zA-Z0-9_]+)\}/', function($matches) use ($varName, $keyName) {
194:                                 if ($matches[1] == $varName) {
195:                                     $str = "<?php echo \${$varName};?>";
196:                                 } elseif ($matches[1] == $keyName) {
197:                                     $str = "<?php echo \${$keyName};?>";
198:                                 } else {
199:                                     $str = $matches[0];
200:                                 }
201:                                 return $str;
202:                             }, $matches[3]);
203:                     $str .= '<?php }?>';
204:                     return $str;
205:                 }, $content);
206: 
207: 
208:         //transfrom variable
209:         $content = preg_replace_callback('/\{\$([\.a-zA-Z0-9_\[\]]+)\}/i', function($matches) {
210:                     $str = '<?php echo $this->varList->';
211:                     $str .= str_replace('.', '->', $matches[1]);
212:                     $str .= ';?>';
213:                     return $str;
214:                 }, $content);
215: 
216: 
217:         //transfrom define variable which is not controller set
218:         $content = preg_replace('/\{set\s+\$([\.a-zA-Z0-9_\x7f-\xff\[\]]+)\s*=\s*(.*)\}/i', "<?php \$this->varList->$1=$2;?>", $content);
219: 
220:         //transfrom if statement
221:         $content = preg_replace_callback('/\{if\s+([^\{^\}]+)\}/i', function($matches) {
222:                     $matches[1] = preg_replace('/\$([\[\]a-zA-Z0-9_\x7f-\xff]+)/i', '$this->varList->$1', $matches[1]);
223:                     $matches[1] = str_replace('.', '->', $matches[1]);
224:                     return "<?php if($matches[1]) { ?>";
225:                 }, $content);
226:         $content = preg_replace_callback('/\{elseif\s+([^\}\{]+)\}/i', function($matches) {
227:                     $matches[1] = preg_replace('/\$([\[\]a-zA-Z0-9_\x7f-\xff]+)/i', '$this->varList->$1', $matches[1]);
228:                     $matches[1] = str_replace('.', '->', $matches[1]);
229:                     return "<?php } elseif({$matches[1]}){ ?>";
230:                 }, $content);
231:         $content = str_replace('{else}', '<?php } else { ?>', $content);
232:         $content = str_replace('{/if}', '<?php } ?>', $content);
233: 
234:         //import other template file
235:         $content = preg_replace('/\{inc\s+(\w+)\}/i', '<?php $this->importFile("$1"); ?>', $content);
236: 
237:         //transfrom invoke php function and echo return value
238:         $content = preg_replace_callback('/\{func\s+([a-zA-Z_\d]+)\((.*)\)\}/i', function ($matches) {
239:                     $matches[2] = str_replace('.', '->', $matches[2]);
240:                     $matches[2] = preg_replace('/\$([\[\]a-zA-Z0-9_\x7f-\xff]+)/i', '$this->varList->$1', $matches[2]);
241:                     return "<?php echo {$matches[1]}({$matches[2]});?>";
242:                 }, $content);
243: 
244:         //clean the whitespace from beginning and end of line and html comment
245:         $content = preg_replace('/^\s*|\s*$|<!--.*-->|[\n\t\r]+/m', '', $content);
246: 
247:         FileObject::saveContent($transfromFile, $content);
248:     }
